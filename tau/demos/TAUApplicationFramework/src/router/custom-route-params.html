Using params in routes


var router = tau.engine.getRouter();
router.addRoute("page-params/:param1/:param2", function (deferred, param1, param2) {
	deferred.resolve(
			'<div data-role="page">' +
			'<div>param1: <strong>' + param1 + '</strong></div>' +
			'<div>param2: <strong>' + param2 + '</strong></div>' +
			'</a>'
	);
});
